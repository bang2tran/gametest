const bird = {

    x: 50,

    y: 150,

    w: 30,

    h: 30,

    radius: 15,

    speed: 0,

    jump: GAME_CONFIG.jumpStrength,

    

    draw: function() {

        ctx.fillStyle = '#FF6584'; // Bird color

        ctx.beginPath();

        ctx.arc(this.x + this.radius, this.y + this.radius, this.radius, 0, Math.PI*2);

        ctx.fill();

        

        // Eye

        ctx.fillStyle = 'white';

        ctx.beginPath();

        ctx.arc(this.x + this.radius + 5, this.y + 5, 6, 0, Math.PI*2);

        ctx.fill();

        ctx.fillStyle = 'black';

        ctx.beginPath();

        ctx.arc(this.x + this.radius + 7, this.y + 5, 2, 0, Math.PI*2);

        ctx.fill();

        

        // Wing

        ctx.fillStyle = '#C24D65';

        ctx.beginPath();

        ctx.ellipse(this.x + 5, this.y + 18, 8, 5, 0.2, 0, Math.PI*2);

        ctx.fill();

    },

    

    update: function() {

        if (!gamePlaying || isPaused) return;

        this.speed += GAME_CONFIG.gravity;

        this.y += this.speed;

        

        // Floor collision

        if (this.y + this.h > canvas.height) {

            this.y = canvas.height - this.h;

            gameOver();

        }

        // Ceiling collision

        if (this.y < 0) {

            this.y = 0;

            this.speed = 0;

        }

    },

    

    flap: function() {

        this.speed = this.jump;

    },

    

    reset: function() {

        this.y = 150;

        this.speed = 0;

    }

            ctx.fillRect(p.x - 2, canvas.height - p.bottom, p.w + 4, 20); // Bottom cap


            // Question Mark

            if (p.isQuestion) {

                ctx.fillStyle = '#FFF';

                ctx.font = 'bold 30px Arial';

                ctx.fillText("?", p.x + 15, p.top - 40);

                ctx.fillText("?", p.x + 15, canvas.height - p.bottom + 50);

            }

        }

    },

    

    update: function() {

        if (!gamePlaying || isPaused) return;

        

        // Add new pipe

        if (frames % Math.floor(GAME_CONFIG.pipeDistance / GAME_CONFIG.gameSpeed * 10) === 0) { // Approx logic

             // Better logic: based on distance

        }

        

        // Simplified generation logic based on x position of last pipe

        if (this.items.length === 0 || (canvas.width - this.items[this.items.length - 1].x >= GAME_CONFIG.pipeDistance)) {

            this.addPipe();

        }


        for(let i = 0; i < this.items.length; i++) {

            let p = this.items[i];

            p.x -= GAME_CONFIG.gameSpeed;

            

            // Collision Detection

            // 1. Horizontal overlap

            if (bird.x + bird.w > p.x && bird.x < p.x + p.w) {

                // 2. Vertical overlap (Hit top or bottom pipe)

                if (bird.y < p.top || bird.y + bird.h > canvas.height - p.bottom) {

                     gameOver();

                }

            }

            

            // Score & Question Logic

            if (p.x + p.w < bird.x && !p.passed) {

                p.passed = true;

                score++;

                updateScore();

                

                if (p.isQuestion) {

                    triggerQuestion();

                }

            }

            

            // Remove off-screen pipes

            if (p.x + p.w < 0) {

                this.items.shift();

                i--;

            }

        }

    },

    

    addPipe: function() {

        // Count how many pipes generated to determine if next is question

        const pipeCount = this.generatedCount || 0;

        this.generatedCount = pipeCount + 1;

        

        const isQ = (this.generatedCount % GAME_CONFIG.questionInterval === 0);

        

        let gap = GAME_CONFIG.pipeGap;

        // Random Y position for gap

        let minHeight = 50;

        let maxPos = canvas.height - minHeight - gap;

        let topHeight = Math.floor(Math.random() * (maxPos - minHeight + 1) + minHeight);

        let bottomHeight = canvas.height - gap - topHeight;

        

        this.items.push({

            x: canvas.width,

            w: 50,

            top: topHeight,

            bottom: bottomHeight,

            passed: false,

            isQuestion: isQ

        });

    },

    

    reset: function() {

        this.items = [];

        this.generatedCount = 0;

    }

};


/**

 * GAME LOGIC CONTROLLERS

 */


function init() {

    resizeCanvas();

    window.addEventListener('resize', resizeCanvas);

    

    // UI Event Listeners

    document.getElementById('btn-start').addEventListener('click', startGame);

    document.getElementById('btn-restart').addEventListener('click', resetGame);

    document.getElementById('btn-reset-header').addEventListener('click', resetGame);

    document.getElementById('btn-teacher-mode').addEventListener('click', toggleTeacherPanel);

    document.getElementById('btn-close-teacher').addEventListener('click', toggleTeacherPanel);

    document.getElementById('btn-save-cfg').addEventListener('click', saveConfig);

    document.getElementById('btn-export').addEventListener('click', exportResult);

    

    // Input Listeners

    window.addEventListener('keydown', (e) => {

        if (e.code === 'Space') {

            e.preventDefault(); // Stop scrolling

            if (gamePlaying && !isPaused) bird.flap();

            if (!gamePlaying && document.getElementById('start-screen').style.display !== 'none') startGame();

        }

    });

    

    canvas.addEventListener('mousedown', (e) => {

        if (gamePlaying && !isPaused) bird.flap();

    });

    

    canvas.addEventListener('touchstart', (e) => {

        e.preventDefault();

        if (gamePlaying && !isPaused) bird.flap();

    }, {passive: false});


    loop();

}


function resizeCanvas() {

    // Canvas resolution matches displayed size for sharpness

    const container = document.getElementById('game-container');

    canvas.width = container.clientWidth;

    canvas.height = container.clientHeight;

}


function loop() {

    ctx.clearRect(0, 0, canvas.width, canvas.height);

    

    // Draw Background (Simple Parallax effect implied)

    ctx.fillStyle = '#E0EAFC'; 

    ctx.fillRect(0,0, canvas.width, canvas.height);

    

    // Draw Clouds (Procedural simple clouds)

    ctx.fillStyle = 'rgba(255,255,255,0.7)';

    for(let i=0; i<5; i++) {

        let x = ((frames * 0.5) + (i * 200)) % (canvas.width + 200) - 100;

        let y = 50 + (i * 30);

        ctx.beginPath();

        ctx.arc(x, y, 30, 0, Math.PI*2);

        ctx.arc(x+40, y, 40, 0, Math.PI*2);

        ctx.arc(x+80, y, 30, 0, Math.PI*2);

        ctx.fill();

    }


    pipes.update();

    pipes.draw();

    

    bird.update();

    bird.draw();

    

    // Ground

    ctx.fillStyle = '#3D4F7D';

    // ctx.fillRect(0, canvas.height - 20, canvas.width, 20);

    

    frames++;

    requestAnimationFrame(loop);

}


function startGame() {

    currentLevel = document.getElementById('level-select').value;

    selectedQuestions = [...GAME_CONFIG.questions[currentLevel]]; // Clone

    

    document.getElementById('start-screen').classList.add('hidden');

    document.getElementById('game-over-screen').classList.add('hidden');

    document.getElementById('question-modal').classList.add('hidden');

    

    bird.reset();

    pipes.reset();

    score = 0;

    frames = 0;

    updateScore();

    

    gamePlaying = true;

    isPaused = false;

    

    startTime = Date.now();

    if(timerInterval) clearInterval(timerInterval);

    timerInterval = setInterval(updateTimer, 1000);

}


function resetGame() {

    gamePlaying = false;

    isPaused = false;

    if(timerInterval) clearInterval(timerInterval);

    

    document.getElementById('start-screen').classList.remove('hidden');

    document.getElementById('game-over-screen').classList.add('hidden');

    document.getElementById('question-modal').classList.add('hidden');

    

    document.getElementById('timer-display').innerText = "00:00";

    document.getElementById('score-display').innerText = "0";

}


function gameOver() {

    gamePlaying = false;

    clearInterval(timerInterval);

    

    document.getElementById('final-score-val').innerText = score;

    document.getElementById('final-time-val').innerText = document.getElementById('timer-display').innerText;

    document.getElementById('game-over-screen').classList.remove('hidden');

}


function updateScore() {

    document.getElementById('score-display').innerText = score;

}


function updateTimer() {

    let delta = Math.floor((Date.now() - startTime) / 1000);

    let m = Math.floor(delta / 60).toString().padStart(2, '0');

    let s = (delta % 60).toString().padStart(2, '0');

    document.getElementById('timer-display').innerText = `${m}:${s}`;

}


/**

 * QUESTION SYSTEM

 */

function triggerQuestion() {

    isPaused = true;

    

    // Pick random question

    const qIndex = Math.floor(Math.random() * selectedQuestions.length);

    const qData = selectedQuestions[qIndex];

    

    const modal = document.getElementById('question-modal');

    const qText = document.getElementById('q-text');

    const qAns = document.getElementById('q-answers');

    

    qText.innerText = qData.q;

    qAns.innerHTML = '';

    

    qData.a.forEach((ans, idx) => {

        const btn = document.createElement('div');

        btn.className = 'answer-btn';

        btn.innerText = ans;

        btn.onclick = () => handleAnswer(idx, qData.correct, btn);

        qAns.appendChild(btn);

    });

    

    modal.classList.remove('hidden');

}


function handleAnswer(selectedIndex, correctIndex, btnElement) {

    const allBtns = document.querySelectorAll('.answer-btn');

    

    // Disable all clicks

    allBtns.forEach(b => b.style.pointerEvents = 'none');

    

    if (selectedIndex === correctIndex) {

        btnElement.classList.add('correct');

        score += 2; // Bonus points

        updateScore();

        

        // Auto resume after 1s

        setTimeout(() => {

            document.getElementById('question-modal').classList.add('hidden');

            isPaused = false;

        }, 1000);

    } else {

        btnElement.classList.add('wrong');

        allBtns[correctIndex].classList.add('correct'); // Show correct one

        

        // Resume after 2s so they can see error

        setTimeout(() => {

            document.getElementById('question-modal').classList.add('hidden');

            isPaused = false;

        }, 2000);

    }

}


/**

 * TEACHER MODE & UTILS

 */

function toggleTeacherPanel() {

    const panel = document.getElementById('teacher-panel');

    if (panel.classList.contains('open')) {

        panel.classList.remove('open');

    } else {

        let pass = prompt("Nhập mật khẩu Teacher Mode:");

        if (pass === GAME_CONFIG.passcode) {

            // Load current configs to inputs

            document.getElementById('cfg-speed').value = GAME_CONFIG.gameSpeed;

            document.getElementById('cfg-gravity').value = GAME_CONFIG.gravity;

            document.getElementById('cfg-gap').value = GAME_CONFIG.pipeGap;

            panel.classList.add('open');

        } else if (pass !== null) {

            alert("Mật khẩu sai!");

        }

    }

}


function saveConfig() {

    GAME_CONFIG.gameSpeed = parseFloat(document.getElementById('cfg-speed').value);

    GAME_CONFIG.gravity = parseFloat(document.getElementById('cfg-gravity').value);

    GAME_CONFIG.pipeGap = parseInt(document.getElementById('cfg-gap').value);

    alert("Đã lưu cài đặt! Reset game để áp dụng.");

    document.getElementById('teacher-panel').classList.remove('open');

}
